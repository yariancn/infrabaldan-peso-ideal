const SHEET_ID = '1N2gx4AR99e6x5TtmnuLSbhj8tn03-yYr1oim40LSCMY'; function doGet(e) { if (!e || !e.parameter) return ContentService.createTextOutput("Error: Acceso manual").setMimeType(ContentService.MimeType.TEXT); if (e.parameter.action === "obtenerClientes") { const ss = SpreadsheetApp.openById(SHEET_ID); const dataC = ss.getSheetByName('Clientes').getDataRange().getValues(); const dataA = ss.getSheetByName('Avances').getDataRange().getValues(); const clientes = dataC.slice(1).map(r => { const id = r[0]; const avances = dataA.slice(1).filter(a => a[0] == id).map(a => ({ fecha: a[1], peso: a[2], grasa: a[3] })); return { id, nombre: r[1], sexo: r[2], edad: r[3], altura: r[4], peso: r[5], grasa: r[6], grasaObjetivo: r[7], fechaInicio: r[8], avances }; }); return ContentService.createTextOutput(JSON.stringify(clientes)).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); if (data.action === "guardarCliente") { const sheet = ss.getSheetByName('Clientes'); const c = data.cliente; sheet.appendRow([c.id, c.nombre, c.sexo, c.edad, c.altura, c.peso, c.grasa, "", c.fechaInicio]); } else if (data.action === "agregarAvance") { const sheet = ss.getSheetByName('Avances'); const av = data.avance; sheet.appendRow([av.id, av.fecha, av.peso, av.grasa]); } return ContentService.createTextOutput(JSON.stringify({success: true})).setMimeType(ContentService.MimeType.JSON); }
